#!/bin/bash
#
# Title:         Unattended Workstation Demo
# Author:        Your Name
# Version:       1.0
# Target:        Windows
# Description:   Opens a browser to a security awareness page to demonstrate
#                the dangers of leaving workstations unattended

# Set the LED to show the payload is running (Yellow)
LED Y SOLID

# Make sure we're a keyboard
ATTACKMODE HID

# Short delay to let system recognize the device
QUACK DELAY 1000

# Open PowerShell with Windows key + R
QUACK GUI r
QUACK DELAY 500
QUACK STRING powershell
QUACK ENTER
QUACK DELAY 1000

# Start a browser and load the security warning page
QUACK STRING Start-Process "http://18.132.143.131/Unattended_Workstation/Unattended_WorkStation_Demo2.html"
QUACK ENTER

# Short delay to let browser open
QUACK DELAY 3000

# Press F11 to go fullscreen
QUACK F11

# Set LED to show completion (Green)
LED G SOLID

# Payload complete
